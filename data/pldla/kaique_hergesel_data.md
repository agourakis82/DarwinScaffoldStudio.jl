## 8. Observações da Degradação

1. **Sem perda de massa significativa** até 90 dias (scaffold mantém estrutura)
2. **Perda de Mw ~87%** em 90 dias (degradação molecular intensa)
3. **Formação de estruturas cristalinas** a partir de 60 dias
4. **Aumento de porosidade** durante degradação
5. **Redução de Tg** com tempo de degradação

### Parâmetros para Darwin Scaffold Studio:

# PLDLA Properties
const PLDLA_PROPERTIES = (
    # Thermal
    Tg = 54.0,  # °C (puro, tempo 0)
    Tg_TEC1 = 49.0,  # °C
    Tg_TEC2 = 46.0,  # °C
    Tmax_degradation = 340.0,  # °C
    
    # Molecular Weight (kg/mol)
    Mw_initial = 94.432,
    Mn_initial = 51.285,
    PDI_initial = 1.84,
    
    # Degradation kinetics (hydrolytic, PBS pH 7.4, 37°C)
    Mw_loss_30d = 0.441,  # 44.1% loss
    Mw_loss_60d = 0.620,  # 62.0% loss  
    Mw_loss_90d = 0.875,  # 87.5% loss
    
    # First-order degradation rate constant (estimated)
    k_hydrolysis = 0.023,  # 1/day (from Mw decay)
    
    # Morphometry (3D printed)
    porosity = 0.395,  # 39.5%
    pore_size_mean = 120.3,  # μm
    pore_size_median = 18.7,  # μm

    # Application
